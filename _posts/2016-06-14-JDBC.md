# lesson 13 Transaction 的处理 重点

Java 里面的DML语句是自动提交的，所以需要先将自动提交属性设置为false。  

~~~java
conn.setAutoCommit(false);
...
conn.commit();
// 恢复现场
conn.setAutoCommut(true);

// 涉及事务，捕捉到异常后需要第一时间rollback（）
conn.rollback();
conn.setAutoCommit(true);
~~~

~~~java
import java.sql.*;
public class TransactionTest {
	
	public static void main(String[] args) {
		Connection conn = null;
		Statement stmt = null;
		
		try {
			Class.forName("oracle.jdbc.driver.OracleDriver");
			conn = DriverManager.getConnection(
				"jdbc:oracle:thin:@192.168.1.105:1521:break", "c##limbo", "tiger");
			stmt = conn.createStatement();
			
			//事物处理部分
			conn.setAutoCommit(false);
			stmt.addBatch("insert into dept2 values(71, 'Boy', 'nj')");
			stmt.addBatch("insert into dept2 values(72, 'Boy', 'nj')");
			stmt.addBatch("insert into dept2 values(73, 'Boy', 'nj')");
			stmt.executeBatch();
			conn.commit();
			conn.setAutoCommit(true);
			
			
		} catch(ClassNotFoundException e) {
			e.printStackTrace();
		} catch(SQLException e) {
			e.printStackTrace();
			try {
				if(conn != null) {
					// 注意回滚
					conn.rollback(); 
					conn.setAutoCommit(true);
				}
			} catch(SQLException k) {
				e.printStackTrace();
			}
		} finally {
			try {
				if(stmt != null) {
					stmt.close();
					stmt = null;
				}
				if(conn != null) {
					conn.close();
					conn = null;
				}
			} catch(SQLException e) {
				e.printStackTrace();
			}
		}
	}
}
~~~

# lesson 14 处理可滚动的结果集

操作结果集的指针

~~~
ResultSet.TYPE_SCROLL_INSENSITIVE,  
ResultSet.CONCUR_READ_ONLY);
~~~

~~~java
import java.sql.*;

public class TestScroll {
	public static void main(String args[]) {

		try {
			new oracle.jdbc.driver.OracleDriver();
			String url = "jdbc:oracle:thin:@192.168.0.1:1521:SXT";
			Connection conn = DriverManager.getConnection(url, "scott", "tiger");
			Statement stmt = conn.createStatement(
					ResultSet.TYPE_SCROLL_INSENSITIVE,
					ResultSet.CONCUR_READ_ONLY);
			ResultSet rs = stmt.executeQuery("select * from emp order by sal");
			rs.next();
			System.out.println(rs.getInt(1));
			rs.last();
			System.out.println(rs.getString(1));
			System.out.println(rs.isLast());
			System.out.println(rs.isAfterLast());
			System.out.println(rs.getRow());
			rs.previous();
			System.out.println(rs.getString(1));
			rs.absolute(6);
			System.out.println(rs.getString(1));
			rs.close();
			stmt.close();
			conn.close();
		} catch (SQLException e) {
			e.printStackTrace();
		}
	}
}

~~~
